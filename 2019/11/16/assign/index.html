# 变量的解构赋值

之前对Es6解构赋值这里理解很模糊,查阅了一些资料,做个总结。

# 数组

// 普通写法
let nameArr = [‘小a', '小b', '小c']; 
let name1 = nameArr[0]; 
let name2 = nameArr[1]; 
let name3 = nameArr[2]; 

// 解构写法 
let [name1, name2, name3] = [‘小a', '小b', '小c']; 
console.log(name1, name2, name3);

# 对象


// 写法1
 let {name, age, sex} = {name: ‘小a', age: 66, sex: ‘man'}; 
console.log(name, age, sex);

// 写法2： 解构重命名
 let {name: XName, age: XAge, sex: XSex} = {name: '小a ', age: 22, sex: '公'};
console.log(XName, XAge, XSex);

// 写法3： 可以设置默认值
 let {name, age, sex = ‘女’} = {name: '小d', age: 33};
 console.log(sex); 

// 写法4：省略解构 
let [, , sex] = [‘小K’, 1, ‘男 ']; 
console.log(sex);

# 字符串

const [a, b, c, d, e] = 'hello'; 
a // "h”
b // "e”

d // "l”

function randomSort(arr) {
    if (toString.call(arr) !== '[object Array]') {
        console.error('请传入一个数组')
    } else {
        let randomIndex, length = arr.length;
        for (let i = 0; i < length; i++) {
         //生成一个原数组长度的下标随机数
        randomIndex = Math.floor(Math.random() * length);

        //常规替换数组内元素位置的方法
        arr.splice(randomIndex, 1, ...arr.splice(i, 1, arr[randomIndex]))
     

        //利用Es6解构赋值随机交换数组内元素的位置
        [arr[randomIndex] , arr[i]] = [arr[i], arr[randomIndex]];
 }
        return arr
    }
}
resultSix = randomSort(testArr);
console.log(resultSix)
